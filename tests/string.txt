function str.reverse(s)
	x = str.buffer()
	s = s.chars()
	for i=len(s)-1,-1,-1 do
		x.write(s[i])
	end
	print (parent(x))
	return x.value()
end
assert("abc", str.from("cba").reverse())
assert("あbc", str.from("cbあ").reverse())
